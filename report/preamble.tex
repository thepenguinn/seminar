\newcommand\alertCaution[1] {
    \begin{tcolorbox}[
            coltitle = colorAlertCaution,
            colbacktitle = colorAlertBgCaution,
            colback = colorAlertBgCaution,
            colframe = colorAlertBgCaution,
            title=Caution,
            fonttitle=\bfseries,
            detach title,
        ]
        \begin{minipage}[t]{0.18\textwidth}
            \begin{flushleft}
                \tcbtitle
            \end{flushleft}
        \end{minipage}
        \begin{minipage}[t]{0.8\textwidth}
            #1
        \end{minipage}
    \end{tcolorbox}
}
